  • repeat the experiment by different mean (using a function)
norm_df =
  function(sample_size, sigma, mean) {
    sample_size = n
    sigma = sigma
    
    list_data = vector("list", length = 5000)
    
    for (i in 1:5000) {
      list_data[[i]] = rnorm(n = sample_size, mean = mean, sd = sigma)
      
    }
    
    result_df =
      expand_grid(sample_size = n,
                  iter = 1:5000) |>
      mutate(
        estimate_df = map(list_data, t.test),
        estimate_df_2 = map(estimate_df, broom::tidy)
      ) |>
      unnest(estimate_df_2) |>
      mutate(mean = mean) |> 
      select(mean, estimate, p.value)
  }
  • Loop the function
outcome_df =
